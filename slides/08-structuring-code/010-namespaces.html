Namespaces

Namespaces can be used to organize and group the functionality on the global object.


// TypeScript
namespace validation {
    export function firstName() {
        // Todo
    }
    export function lastName() {
        // Todo
    }
}

class RegisterUser {
    validate = validation.firstName();
}


// JavaScript
var validation;
(function (validation) {
    function firstName() {
        // Todo
    }
    validation.firstName = firstName;
    function lastName() {
        // Todo
    }
    validation.lastName = lastName;
})(validation || (validation = {}));

class RegisterUser {
    constructor() {
        this.validate = validation.firstName();
    }}
