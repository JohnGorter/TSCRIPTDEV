Varargs


//TypeScript

function buyNewCar(brand: string,
    ...features: string[]) {
    console.log(`${brand}
      with features: ${features.join(', ')}`);
}

buyNewCar('Tesla',
    'Leather seats', 'Blue painting');

// => "Tesla with features:
//    Leather seats, Blue painting"


//Javascript

function buyNewCar(brand) {
    var features = [];
    for (var _i = 1; _i <
        arguments.length; _i++) {
        features[_i - 1] = arguments[_i];
    }
    console.log(brand +
    'with features: ' + features.join(', '));
}
buyNewCar('Tesla', 'Leather seats', 'Blue painting');
  • Varargs parameters should be placed after required parameters
