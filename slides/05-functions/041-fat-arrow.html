Fat arrow functions

this keyword is (kind of) broken in ECMAScript. 

var car = {
    all: ['Model S', 'Golf'],
    selected: '',

    select(carType: string) {
        this.all.forEach(function (t) {
            if (t === carType) {
                this.selected = t;
            }
         })
    }
}

car.select('Model S')
car.selected; // => Empty string
window['selected']; // => "Model S" the `this` points to window.
