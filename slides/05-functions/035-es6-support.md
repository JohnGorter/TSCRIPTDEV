### Object matching

![es6](resources/es6.png)<!-- .element class="emblem"  -->

```typescript
function print({name: string, age: number}){
    console.log(`${name} is ${age}`);
}
print({name: 'Obiwan', age: 68});

function han(){
    return {
        name: 'Han Solo'
    }
}
const { name: hansName } = han();
hansName // => 'Han Solo'
```

### Array Matching

```typescript
const list = [ 1, 2, 3 ];
const [ a, , b ] = list;
a // => 1
b // => 3
```
